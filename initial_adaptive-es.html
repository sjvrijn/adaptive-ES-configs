def loadfile(fname, max_budget):
    """
    Load a file

    :param fname:       The name of the file to retrieve the data from
    :param max_budget:  Maximum available budget for this optimization run
    :return:            Data from the given file as numpy float array
    """

    fitnesses = np.empty(max_budget)
    with open(fname, 'r') as f:
        next(f)  # Skip the first line which only contains header information
        prev_idx = 0
        prev_fitness = np.inf
        for line in f:
            idx, fitness = line.split(' ')[:2]        # First two columns of the file
            idx = int(idx)
            fitness = float(fitness)
            fitnesses[prev_idx:idx] = prev_fitness  # Fill the array up to the new idx with previous value
            prev_idx, prev_fitness = idx, fitness
        
        # Fill the rest of the array with the last value
        fitnesses[prev_idx:] = prev_fitness
        
    return fitnesses

This next step will take several minutes use a lot of RAM (few GB's).

All files will be loaded, and for each file an entire array of length max_budget will be allocated.

def determine_steepnesses(data):
    steepnesses = np.array([0]*len(steepness_sections), dtype=np.float)
    prev_idx = 0
    for idx, stp_sec in enumerate(steepness_sections):
        below_target = data < stp_sec
        indices = np.argwhere(below_target)
        if len(indices) > 0:
            dx = 1
            dy = min(indices) - prev_idx
            steepness = dx/dy
            steepnesses[idx] = steepness[0]
            prev_idx = min(indices)
        else:
            break
    
    return steepnesses
